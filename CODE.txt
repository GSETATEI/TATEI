// Clasifica un registro por su % AVANCE numérico, ignorando el STATUS del CSV
// Si el registro no tiene un % AVANCE válido, mantiene el STATUS original del CSV
function clasificar(d) {
  const pct = parseInt(d.AVANCE) || 0;
  for (const n of NIVEL_CONFIG) {
    if (pct >= n.min && pct < n.max) return n.key;
    if (n.max === 999 && pct >= n.min) return n.key;
  }
  return d.STATUS; // fallback al STATUS del CSV
}

function getColor(status) {
  const cfg = NIVEL_CONFIG.find(n => n.key === status);
  return cfg ? cfg.color : '#8C96A8';
}
function getBg(status) {
  const cfg = NIVEL_CONFIG.find(n => n.key === status);
  return cfg ? cfg.bg : 'rgba(140,150,168,0.90)';
}

function buildMarkers(filteredData) {
  if (clusterGroup) map.removeLayer(clusterGroup);
  markers = [];

  // Plain layer group — no clustering, all markers visible at every zoom
  clusterGroup = L.layerGroup();

  filteredData.forEach(d => {
    const marker = L.marker([d.Y, d.X], {
      icon: createIcon(d),
      data: d
    });

    marker.bindTooltip(`${d.CCT} — ${clasificar(d)} (${d.AVANCE})`, {
      className: 'custom-tooltip',
      direction: 'top',
      offset: [0, -30],
      opacity: 1
    });

    const popupContent = buildPopupHTML(d);
    marker.bindPopup(popupContent, {
      maxWidth: 260,
      closeButton: true,
      className: ''
    });

    marker.on('click', function() {
      if (selectedMarker && selectedMarker !== this) {
        selectedMarker.setIcon(createIcon(selectedMarker.options.data));
      }
      this.setIcon(createIconSelected(this.options.data));
      selectedMarker = this;
      showDetail(d);
      highlightListItem(d.CCT);
    });

    markers.push(marker);
    clusterGroup.addLayer(marker);
  });

  map.addLayer(clusterGroup);
}

function buildResultsList(filteredData) {
  const list = document.getElementById('results-list');
  const noRes = document.getElementById('no-results');
  const header = document.getElementById('results-header');

  list.querySelectorAll('.result-item').forEach(el => el.remove());

  if (filteredData.length === 0) {
    noRes.style.display = 'block';
    return;
  }
  noRes.style.display = 'none';

  filteredData.forEach(d => {
    const nivel = clasificar(d);
    const color = getColor(nivel);
    const item = document.createElement('div');
    item.className = 'result-item';
    item.dataset.cct = d.CCT;
    item.innerHTML = `
      <div class="ri-dot" style="background:${color};"></div>
      <span class="ri-cct">${d.CCT}</span>
      <span class="ri-proj">${d.PROYECTO}</span>
      <span class="ri-status" style="background:${color}1A;color:${color};">${nivel}</span>
    `;
    item.addEventListener('click', () => {
      const marker = markers.find(m => m.options.data.CCT === d.CCT);
      if (marker) {
        map.setView([d.Y, d.X], 14, { animate: true });
        if (selectedMarker && selectedMarker !== marker) {
          selectedMarker.setIcon(createIcon(selectedMarker.options.data));
        }
        marker.setIcon(createIconSelected(d));
        selectedMarker = marker;
        marker.openPopup();
        showDetail(d);
        highlightListItem(d.CCT);
      }
    });
    list.appendChild(item);
  });
}

let statusChart = null;

  const total = filteredData.length;

  // ── Conteo por nivel calculado desde % AVANCE ──
  const counts = {};
  NIVEL_CONFIG.forEach(n => counts[n.key] = 0);

    if (counts[k] !== undefined) counts[k]++;
  });

  const labels = NIVEL_CONFIG.map(n => n.label.toUpperCase());
  const values = NIVEL_CONFIG.map(n => counts[n.key]);
  const colors = NIVEL_CONFIG.map(n => n.color);

  // ── Donut ──
  const ctx = document.getElementById('statusChart').getContext('2d');
  if (statusChart) statusChart.destroy();
  statusChart = new Chart(ctx, {
    type: 'doughnut',
    data: { labels, datasets: [{ data: values, backgroundColor: colors, borderWidth: 0, hoverOffset: 4 }] },
    options: {
      cutout: '72%', responsive: false, animation: { duration: 600 },
      plugins: {
        legend: { display: false },
        tooltip: { callbacks: { label: (c) => ` ${c.label}: ${c.raw} (${total > 0 ? Math.round(c.raw/total*100) : 0}%)` } }
      }
    }
  });

  // ── Leyenda donut ──
  const leg = document.getElementById('legend-container');

    badge.style.display = 'none';

  ctrArr.forEach(({ name, avg }) => {
    const nivel = NIVEL_CONFIG.find(n => avg >= n.min && (n.max === 999 ? avg >= n.min : avg < n.max));

    proyCnt[d.PROYECTO]    = (proyCnt[d.PROYECTO]    || 0) + 1;
  });
  const proyArr = Object.keys(proyAvance).map(name => ({
    name, avg: proyCnt[name] > 0 ? Math.round(proyAvance[name]/proyCnt[name]) : 0, n: proyCnt[name]

  const proyEl = document.getElementById('proyectos-bars');
  proyEl.innerHTML = '';
  proyArr.forEach(({ name, avg, n }) => {
    const nivel = NIVEL_CONFIG.find(nv => avg >= nv.min && (nv.max === 999 ? avg >= nv.min : avg < nv.max));

      </div>`;
  });
}

function applyFilters() {
  const proj = document.getElementById('f-proyecto').value;
  const ctr  = document.getElementById('f-contratista').value;
  const cct  = document.getElementById('f-cct').value.trim().toUpperCase();

  // baseData = todos los filtros activos EXCEPTO el de estatus
  const baseData = DATA.filter(d => {
    if (proj && d.PROYECTO !== proj) return false;
    if (ctr  && d.CONTRATISTA !== ctr) return false;
    if (cct  && !d.CCT.includes(cct)) return false;
    return true;
  });

  // filtered = baseData + filtro de nivel (calculado desde % AVANCE)
  const filtered = activeStatus === 'TODOS'
    ? baseData
    : baseData.filter(d => clasificar(d) === activeStatus);

  buildMarkers(filtered);
  buildResultsList(filtered);
  updateStats(filtered, baseData);
  buildCharts(filtered, proj, ctr);
}

function debounce(fn, ms) {
  let t;
  return function(...args) {
    clearTimeout(t);
    t = setTimeout(() => fn.apply(this, args), ms);
  };
}

// Columnas requeridas en el archivo (mayúsculas, sin espacios extra)
const REQUIRED_COLS = ['CCT','PROYECTO','CONTRATISTA','CONTRATO','X','Y'];

function showToast(msg, type = 'ok') {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.className = 'show ' + type;
  clearTimeout(t._timer);
  t._timer = setTimeout(() => { t.className = ''; }, 3500);
}

function rowsToData(rows) {
  // rows: array of objects con claves = encabezados del archivo
  const normalize = k => k.trim().toUpperCase().replace(/\s+/g,'_');
  // Detectar nombres de columnas normalizados
  if (!rows.length) throw new Error('El archivo está vacío.');
  const sample = rows[0];
  const keys = Object.keys(sample);
  const keyMap = {};
  keys.forEach(k => keyMap[normalize(k)] = k);

  // Mapeo flexible: acepta MONTO o VALOR, % AVANCE o %_AVANCE o AVANCE
  const get = (norm, row) => {
    const aliases = {
      'VALOR':   ['VALOR','MONTO','IMPORTE'],
      'AVANCE':  ['%_AVANCE','%AVANCE','AVANCE','%_AVANCE_'],
      'ESTIMADO':['ESTIMADO','MONTO_ESTIMADO'],
      'ACTIVIDADES':['ACTIVIDADES','ACTIVIDAD'],
      'COMENTARIOS':['COMENTARIOS','COMENTARIO','NOTAS'],
      'STATUS':  ['STATUS','ESTATUS','ESTADO'],
    };
    const list = aliases[norm] || [norm];
    for (const a of list) {
      if (keyMap[a]) return String(row[keyMap[a]] ?? '').trim();
    }
    return '';
  };

  // Validar columnas mínimas
  for (const req of REQUIRED_COLS) {
    const aliases = { 'X':['X','LONGITUD','LON','LNG'], 'Y':['Y','LATITUD','LAT'] };
    const list = aliases[req] || [req];
    if (!list.some(a => keyMap[a])) throw new Error(`Columna requerida no encontrada: ${req}`);
  }

  const getCoord = (norm, row) => {
    const aliases = { 'X':['X','LONGITUD','LON','LNG'], 'Y':['Y','LATITUD','LAT'] };
    const list = aliases[norm] || [norm];
    for (const a of list) {
      if (keyMap[a]) return parseFloat(row[keyMap[a]]);
    }
    return NaN;
  };

  return rows
    .filter(row => getCoord('X', row) && getCoord('Y', row))  // descartar sin coords
    .map(row => ({
      CONTRATO:    get('CONTRATO', row)    || '—',
      PROYECTO:    get('PROYECTO', row)    || '—',
      CONTRATISTA: get('CONTRATISTA', row) || '—',
      CCT:         get('CCT', row)         || '—',
      VALOR:       get('VALOR', row)       || '—',
      ESTIMADO:    get('ESTIMADO', row)    || '—',
      AVANCE:      get('AVANCE', row)      || '0%',
      STATUS:      get('STATUS', row)      || '—',
      ACTIVIDADES: get('ACTIVIDADES', row) || '—',
      COMENTARIOS: get('COMENTARIOS', row) || '',
      X: getCoord('X', row),
      Y: getCoord('Y', row),
    }));
}

function loadNewData(records) {
  if (!records.length) throw new Error('No se encontraron registros con coordenadas válidas.');

  // Reemplazar DATA global
  DATA.length = 0;
  records.forEach(r => DATA.push(r));

  // Actualizar contador total en header
  document.getElementById('total-fixed').textContent = DATA.length;

  // Repoblar dropdowns
  const fproy = document.getElementById('f-proyecto');
  const fctr  = document.getElementById('f-contratista');
  // Limpiar manteniendo opción vacía
  while (fproy.options.length > 1) fproy.remove(1);
  while (fctr.options.length  > 1) fctr.remove(1);
  [...new Set(DATA.map(d => d.PROYECTO))].sort().forEach(p => {
    const o = document.createElement('option'); o.value = p; o.textContent = p; fproy.appendChild(o);
  });
  [...new Set(DATA.map(d => d.CONTRATISTA))].sort().forEach(c => {
    const o = document.createElement('option'); o.value = c; o.textContent = c; fctr.appendChild(o);
  });

  // Reset filtros y redibujar
  document.getElementById('f-proyecto').value = '';
  document.getElementById('f-contratista').value = '';
  document.getElementById('f-cct').value = '';
  activeStatus = 'TODOS';
  document.querySelectorAll('.status-pill').forEach(p => p.classList.remove('active'));
  document.querySelector('.status-pill[data-status="TODOS"]').classList.add('active');
  if (selectedMarker) { selectedMarker = null; }
  document.getElementById('selected-panel').style.display = 'none';
  document.getElementById('sp-placeholder-section').style.display = '';

  applyFilters();

  // Centrar mapa en nuevos datos
  const lats = DATA.map(d => d.Y), lons = DATA.map(d => d.X);
  const bounds = L.latLngBounds(
    [Math.min(...lats), Math.min(...lons)],
    [Math.max(...lats), Math.max(...lons)]
  );
  map.fitBounds(bounds, { padding: [40, 40], animate: true });
}

function parseFile(file) {
  const ext = file.name.split('.').pop().toLowerCase();
  const reader = new FileReader();

  reader.onload = e => {
    try {
      let rows;
      if (ext === 'csv') {
        // Parsear CSV manualmente respetando comillas
        const text = e.target.result;
        const lines = text.split(/\r?\n/).filter(l => l.trim());
        const headers = lines[0].split(',').map(h => h.replace(/^"|"$/g,'').trim());
        rows = lines.slice(1).map(line => {
          // Split respetando campos entre comillas
          const vals = [];
          let cur = '', inQ = false;
          for (const ch of line + ',') {
            if (ch === '"') { inQ = !inQ; }
            else if (ch === ',' && !inQ) { vals.push(cur.trim()); cur = ''; }
            else cur += ch;
          }
          const obj = {};
          headers.forEach((h, i) => obj[h] = vals[i] ?? '');
          return obj;
        });
      } else {
        // Excel via SheetJS
        const wb = XLSX.read(e.target.result, { type: 'array' });
        const ws = wb.Sheets[wb.SheetNames[0]];
        rows = XLSX.utils.sheet_to_json(ws, { defval: '' });
      }

      const records = rowsToData(rows);
      loadNewData(records);
      showToast(`✓ ${records.length} registros cargados desde "${file.name}"`, 'ok');
    } catch (err) {
      showToast(`✗ Error: ${err.message}`, 'err');
      console.error(err);
    }
    document.getElementById('drop-overlay').classList.remove('visible');
  };

  reader.onerror = () => showToast('✗ No se pudo leer el archivo.', 'err');

  if (ext === 'csv') reader.readAsText(file, 'UTF-8');
  else reader.readAsArrayBuffer(file);
}
